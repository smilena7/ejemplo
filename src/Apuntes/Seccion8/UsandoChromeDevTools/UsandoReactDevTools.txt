class Counter extends Component {
  state = {
    clicks: 0
  }

  aumentador = () => {
    this.setState(() => ({
      clicks: this.state.clicks + 1
    }))
  }

  disminuidor = () => {
    this.setState(() => ({
      clicks: this.state.clicks - 1
    }))
  }
  render () {
    return (
      <>
      <button onClick={this.aumentador}>
        Increment
      </button>
      <button onClick={this.disminuidor}>
        Decrement
      </button>
      <h1>
        { this.state.clicks }
      </h1>
      </>
    )
  }
}

  }
}
